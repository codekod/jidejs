This demo shows how to use a {@link jidejs/ui/control/TextField} to let users insert text into an application.

require([
    'jidejs/ui/control/TextField', 'jidejs/ui/control/PasswordField',
    'jidejs/ui/control/Label', 'jidejs/ui/control/Button',
    'jidejs/ui/layout/HBox'
], function(TextField, PasswordField, Label, Button, HBox) {
    var name = new TextField();
    var password = new PasswordField();
    // in order to get baseline alignment, we need to wrap the elements in an HBox and change its style
    var box = new HBox({
        spacing: '5px',
        children: [
            new Label({ text: 'Name' }),
            name,
            new Label({ text: 'Password' }),
            password,
            new Button({
                text: 'Submit',
                on: {
                    action: function() {
                        alert("You've entered the following login data: "+name.text+", "+password.text);
                    }
                }
            })
        ]
    });
    // set styles to align the items at baseline
    box.style.set('align-items', 'baseline')
        .set('flex-align', 'baseline')
        .set('-webkit-align-items', 'baseline')
        .set('-moz-box-align', 'baseline').update();
    root.appendChild(box.element);
});

This example shows how a very simple login form could be created using jide.js. In order to get baseline alignment, we've chosen to use a simple {@link jidejs/ui/layout/HBox} for the controls.

Try to listen to changes to the {@link jidejs/ui/control/TextField} or its action event, which is activated when the user presses the 'Enter' key while the {@link jidejs/ui/control/TextField} has the focus.

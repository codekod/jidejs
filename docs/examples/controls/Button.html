This demo shows how to use a {@link jidejs/ui/control/Button}.

// use AMD/require.js to load the Button control
require(['jidejs/ui/control/Button'], function(Button) {
    // create a new Button
    new Button({
        // in this case: Select a DOM element
        element: root.querySelector('#exampleButton'),
        // specify the text that should be shown for the button
        text: 'Hello World!',
        // add the 'action' event handler
        onaction: function() {
            // simply show an alert
            alert('Hello World!');
        }
    });
});

This demo shows how to use a {@link jidejs/ui/control/Button}.

To get started, try to modify the value of the text property and press the Run button.

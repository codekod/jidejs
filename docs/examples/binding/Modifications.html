This demo shows how an alternative to {@link jidejs/base/Observable~computed} by using a higher level API.

    require(['jidejs/base/Observable'], function(Observable) {
    // create two observable value and a computed observable
    var age = Observable(42),
        name = Observable('John'),
        computedToString = Observable.computed(function() {
            return name.get() + ' is '+ age.get() + ' years old.';
        });
    log('Computed property way:');
    log(computedToString.get());

    // an alternative would be to use the `concat` method:
    var fluentToString = name.concat(' is ', age, ' years old');
    log('Fluent API');
    log(fluentToString.get());

    // this utility function appends an entry to our visible log
    function log(msg) {
        root.querySelector('#xlog').appendChild(function() {
            var li = document.createElement('li');
            li.innerHTML = msg;
            return li;
        }());
    }
});

    Even though creating computed observables is easy, jide.js provides a fluent API that could be even easier to use, depending on your use case.
